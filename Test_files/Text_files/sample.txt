#include <bits/stdc++.h>
#include <iostream>
using namespace std;

int countwords(string word){
    word = word+'.';
    int flag,count=0;
    // cout<<word<<endl;
    if(isdigit(word[0]) || isalpha(word[0]))
        flag=1;
    else
        flag=0;

    for(int i=0; i<word.size(); i++){
        if(flag==1){
            if(isalpha(word[i])||isdigit(word[i])){
                flag=1;
            }
            else{
                flag=0;
                count++;
            }
        }
        else{
            if(isalpha(word[i])||isdigit(word[i])){
                flag=1;
                continue;
            }
        }
    }
    return count;
}

int main(){

    string filename,line,word;
    char ch;
    cin>>filename;

    ifstream fin;
    fin.open(filename);

    int numLines=0,numWords=0,numSpace=0,numChar=0,flag=0,num=0;
    while(getline(fin,line)){

        numLines++;
        for(int i=0; i<line.length(); i++){
            ch=line[i];
            numChar++;
            if(isspace(ch)){
                numSpace++;
                numChar--;
            }
        }
        stringstream s(line);

        while (s >> word){
            flag=0;
            for(int i=0; i<word.length(); i++){
                if(!(isalpha(word[i]) || isdigit(word[i])))
                    flag=1;
            }
            if(flag){
                numWords+=countwords(word);
            }
            else{
                numWords++;
            }
        }
    }
    fin.close();

    cout<<numLines<<" " << numWords<<" " <<numSpace+numChar+numLines-1<<" "<< numChar<<" "<< numSpace+numChar+numLines-1<<endl;


    string outputfile;
    outputfile = "Documentstatic-"+filename;

    fstream fout;
    fout.open(outputfile,ios::out);

    fin.open(filename);

    if(!fout){
        return 0;
    }


    fout<<endl<<endl;
    fout<<"Lines - "<<numLines<<endl;
    fout<<"Words - "<<numWords<<endl;
    fout<<"Characters (with spaces) - "<<numSpace+numChar+numLines-1<<endl;
    fout<<"Characters (no spaces) - "<<numChar<<endl;
    fout<<"Bytes - "<<numSpace+numChar+numLines-1<<endl;
    fout<<endl<<endl;

    while (getline(fin,line)){
        fout<<line;
    }
    return 0;
}
